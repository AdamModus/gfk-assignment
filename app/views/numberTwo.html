Write a class (prototype) in JavaScript, with at least one method, and another class which extends the previous class.

Make sure that when an instance is created of the latter, it returns true for an instanceof-check check for both prototypes and that the method on the first prototype can be invoked. 

function Feline() {
   this.purr = function() {
      return 'purrrrr';
   };
}

function Leopard() {
   this.roar = function() {
      return 'ROOOAAARRRRR';
   };
}

Leopard.prototype = new Feline();

var leppy = new Leopard();

console.log('Is it a feline? ' + (leppy instanceof Feline ? 'yes :D' : 'oh no!'));
console.log('Is it a leopard? ' + (leppy instanceof Leopard ? 'yes :3' : 'oh no!'));

console.log(leppy.purr());
console.log(leppy.roar());
